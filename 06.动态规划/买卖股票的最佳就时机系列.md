# 买卖股票的最佳就时机系列

## 121. 买卖股票的最佳时机

```
class Solution {
public:
    int maxProfit(vector<int>& prices) {
        vector<vector<int>>dp(prices.size(), vector<int>(2, 0));
        dp[0][0] = 0;
        dp[0][1] = -prices[0];
        int ret = dp[0][0];
        for(int i = 1;i<prices.size();i++)
        {
            dp[i][0] = max(dp[i-1][0], dp[i-1][1]+prices[i]);
            dp[i][1] = max(dp[i-1][1], -prices[i]);
            if(dp[i][0] > ret)
            {
                ret = dp[i][0];
            }
        }
        return ret;
    }
};
```
